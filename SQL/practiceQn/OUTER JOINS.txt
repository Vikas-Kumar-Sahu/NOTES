RIGHT-OUTER-JOIN 
SYNTAX:
SELECT * FROM EMP E, DEPT D
WHERE E.DEPTNO(+) = D.DEPTNO;

LEFT-OUTER-JOIN 
SYNTAX:
SELECT * FROM EMP E, DEPT D
WHERE E.DEPTNO = D.DEPTNO(+);

FULL-OUTER-JOIN 
SYNTAX:
SELECT * FROM EMP E FULL OUTER JOIN DEPT D
ON E.DEPTNO = D.DEPTNO;


Q2.WAQTD NAME OF EMP EARNING SECOND MAX SAL IN NEW YORK?
SELECT ENAME FROM EMP E, DEPT D
WHERE E.DEPTNO = D.DEPTNO
AND SAL IN (SELECT MAX(SAL)
  FROM EMP WHERE SAL < (SELECT MAX(SAL)
     FROM EMP WHERE DEPTNO IN (SELECT DEPTNO
        FROM DEPT WHERE LOC = 'NEW YORK'))
AND DEPTNO IN (SELECT DEPTNO
        FROM DEPT WHERE LOC = 'NEW YORK'));
