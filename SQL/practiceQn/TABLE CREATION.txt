CREATE TABLE STUDENT
(
	SID NUMBER(3),
	SNAME VARCHAR(25),
	BRANCH VARCHAR(25),
	PERC NUMBER(5,2),
	SUBID NUMBER(2),
	COURSE VARCHAR(2),
	PHNO NUMBER(10),

	CONSTRAINT STUDENTID PRIMARY KEY(SID),
	CONSTRAINT PHONE_NUMBER UNIQUE(PHNO),
	CONSTRAINT PHONE_NUM CHECK(LENGTH(PHNO)=10),
	CONSTRAINT COURSE FOREIGN KEY(SUBID) REFERENCES SUBJECT_DETAILS(SUBID),
	CONSTRAINT COURSE FOREIGN KEY(COURSE) REFERENCES COURSE_DETAILS(COURSE)
);

INSERT INTO STUDENT (SID, SNAME, BRANCH, PERC, SUBID, COURSE, PHNO) 
VALUES (&SID, &SNAME, &BRANCH, &PERC, &SUBID, &COURSE, &PHNO);

	

CREATE TABLE COURSE_DETAILS
(
	COURSE VARCHAR(2),
	COURSE_NAME VARCHAR(25),
	DURATION VARCHAR(20),
	CONSTRAINT COURSE PRIMARY KEY(COURSE)
);

INSERT INTO COURSE_DETAILS (COURSE, COURSE_NAME, DURATION) VALUES (&COURSE, &COURSE_NAME, &DURATION);

CREATE TABLE SUBJECT_DETAILS
(
	SUBID NUMBER(2),
	SUBNAME VARCHAR(25),
	TRAINER VARCHAR(25),
	CONSTRAINT SUBJECT_ID PRIMARY KEY (SUBID)
);

INSERT INTO SUBJECT_DETAILS (SUBID, SUBNAME, TRAINER) VALUES (&SUBID, &SUBNAME, &TRAINER);
