9.WAQTD NAMES OF AN EMP HIRE BEFORE ADAMS MANAGER BUT HIRE AFTER ALLEN'S MANAGER AND EARNING LESS THAN WARD'S MANAGER'S 
MANAGER AND HE MUST NOT BE WORKING IN MARTIN DESIGNATION AND HIRED IN MONTH OF DECEMBER IN NEW YORK OR DALLAS? FORD
SELECT ENAME FROM EMP 
WHERE HIREDATE < (SELECT HIREDATE FROM EMP
                  WHERE EMPNO IN (SELECT MGR FROM EMP 
                                  WHERE ENAME = 'ADAMS'))
AND HIREDATE > (SELECT HIREDATE FROM EMP 
                WHERE EMPNO IN (SELECT MGR FROM EMP
                               WHERE ENAME = 'ALLEN'))
AND SAL < (SELECT SAL FROM EMP
           WHERE EMPNO IN (SELECT MGR FROM EMP 
                           WHERE EMPNO IN (SELECT MGR FROM EMP
                                          WHERE ENAME = 'WARD')))
AND JOB NOT IN (SELECT JOB FROM EMP
                WHERE ENAME = 'MARTIN')
AND HIREDATE LIKE '%DEC%'
AND DEPTNO IN (SELECT DEPTNO FROM DEPT 
               WHERE LOC IN ('NEW YORK','DALLAS'));
ENAME
----------
FORD
