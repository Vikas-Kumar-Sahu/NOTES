Q.3)WAQTD ENAME ,DNAME AND LOC OF EMP WORKING IN SAME DESIGNATION AS THAT OF MILLER HIRED BEFORE ADAMS IN SCOTT DEPT?
SELECT ENAME,LOC,DNAME
FROM EMP,DEPT
WHERE EMP.DEPTNO = DEPT.DEPTNO
AND JOB IN (SELECT JOB FROM EMP
		WHERE ENAME = 'MILLER')

		WHERE ENAME = 'ADAMS')

		WHERE ENAME = 'SCOTT');

Q.10)WAQTD ENAME,DNAME,DEPTNO OF EMP EARNING 3RD MAX SAL?
SELECT ENAME, DNAME, EMP.DEPTNO
FROM EMP, DEPT
WHERE EMP.DEPTNO = DEPT.DEPTNO
AND SAL IN (SELECT MAX(SAL)
		FROM EMP WHERE SAL < (SELECT MAX(SAL)
			FROM EMP WHERE SAL < (SELECT MAX(SAL)
						FROM EMP)));
