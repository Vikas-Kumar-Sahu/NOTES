Q1.WAQTD SELECT 2ND MIN SAL? 950
SELECT MIN(SAL)
FROM EMP
WHERE SAL > (SELECT MIN(SAL)
		FROM EMP);
  MIN(SAL)
----------
       950

Q2.WAQTD NAME OF AN EMP EARNING 2ND MIN SAL? JMAES
SELECT ENAME
FROM EMP
WHERE SAL IN (SELECT MIN(SAL)
		FROM EMP
		WHERE SAL > (SELECT MIN(SAL)
				FROM EMP));
ENAME
----------
JAMES

Q3.WAQTD DEPARTMENT NAME OF AN EMP EARNING THE MAX SAL? ACCOUNTING
SELECT DNAME
FROM DEPT
WHERE DEPTNO IN (SELECT DEPTNO
		FROM EMP
		WHERE SAL IN (SELECT MAX(SAL)
				FROM EMP));
DNAME
--------------
ACCOUNTING
