Q2.WAQTD 10TH MAX SAL?
SELECT SAL
FROM (SELECT ROWNUM AS SLNO, SAL
	FROM (SELECT DISTINCT SAL
		FROM EMP ORDER BY SAL DESC))
WHERE SLNO = 10;

----------
